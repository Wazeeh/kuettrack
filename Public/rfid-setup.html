Setup Guide
Step-by-Step Setup
Follow these steps to get your bike station running in under 30 minutes.
✓
1. Backend Setup — Verify API is Running
Make sure your KuetTrack backend (server.js) is deployed and accessible. The ESP-32 needs to reach http://YOUR_SERVER_IP:5000/api/auth/user-by-rfid/:uid over WiFi.
💡 For local testing, make sure the ESP-32 and your laptop are on the same WiFi network. Use your LAN IP (e.g. 192.168.1.100), not localhost.
2
2. Wire the Hardware
Connect the MFRC522 to the ESP-32 via SPI. See the Wiring tab for the complete pin diagram.
⚠️ The MFRC522 operates at 3.3V. Do NOT connect to 5V or you will damage the module!
3
3. Install Arduino Libraries
In Arduino IDE, install these libraries via Sketch → Include Library → Manage Libraries:
MFRC522by GithubCommunity (v1.4.10+)
ArduinoJsonby Benoit Blanchon (v6.x)
LiquidCrystal I2Cby Frank de Brabander (for LCD)
ESP32Servoby Kevin Harrington (for lock servo)
Also in File → Preferences → Additional Boards URLs add: https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package_esp32_index.json then install esp32 by Espressif in Board Manager.
4
4. Upload Arduino Code
Go to the Arduino Code tab, copy the full sketch, fill in your WiFi credentials and server IP, then upload to the ESP-32. Open Serial Monitor at 115200 baud to see logs.
5
5. Assign RFID Cards in Admin Panel
Go to Admin Dashboard → RFID Cards section. Scan a card UID using the Live Tester below or read it from Serial Monitor, then assign it to a user account.
6
6. Test Authentication
Tap the assigned RFID card on the reader. The ESP-32 should contact the API, verify the UID, and return user information with access status. Use the Live Tester to simulate scans from your browser.

REST API Reference
RFID & Auth Endpoints
All endpoints the ESP-32 and admin panel use. The ESP-32 primarily uses the user-by-rfid endpoint.
GET /api/auth/user-by-rfid/:uid
🔲 Used by ESP-32. Verifies an RFID card UID against the database. Returns user info if authorized. No auth token needed — the UID is the credential.
// ESP-32 HTTP call: GET http://192.168.1.100:5000/api/auth/user-by-rfid/A3:4F:2B:11 // ✅ Success Response (200): { "authorized": true, "user": { "id": "64abc...", "firstName": "Rahul", "lastName": "Ahmed", "plan": "monthly", "walletBalance": 250 } } // ❌ Not Found (404): { "authorized": false, "message": "RFID card not recognized." }
PATCH /api/auth/rfid
🛡️ Admin only. Assigns an RFID card UID to a user account. Requires admin JWT token in Authorization header.
// Request (Admin JWT required): PATCH http://server:5000/api/auth/rfid Authorization: Bearer <admin_jwt_token> Content-Type: application/json { "userId": "64abc...", "rfidUid": "A3:4F:2B:11" } // ✅ Response: { "message": "RFID card assigned successfully.", "user": { ... } }
POST /api/rides/start
🚲 Used by ESP-32 after auth. Logs the start of a ride session after successful RFID authentication.
// Request: POST http://server:5000/api/rides/start Content-Type: application/json { "rfidUid": "A3:4F:2B:11", "stationId": "STATION_01", "bikeId": "VT-023" } // ✅ Response: { "message": "Ride started.", "rideId": "ride_abc123", "user": { "firstName": "Rahul", "plan": "monthly" } }
POST /api/rides/end
🏁 Used by ESP-32 on second tap. Ends an active ride session, calculates fare, and deducts from wallet.
// Request: POST http://server:5000/api/rides/end Content-Type: application/json { "rfidUid": "A3:4F:2B:11", "rideId": "ride_abc123", "stationId": "STATION_02" } // ✅ Response: { "message": "Ride ended.", "duration": "00:14:32", "fare": 30, "newBalance": 220 }

Arduino / C++ 
// ================================================================
//  KuetTrack RFID Station — ESP-32 Firmware
//  Hardware: ESP-32 + MFRC522 + LCD I2C + LEDs + Buzzer + Servo
//  Author: KuetTrack IoT System
// ================================================================

#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <SPI.h>
#include <MFRC522.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <ESP32Servo.h>

// ── WiFi Credentials ─────────────────────────────────────────
const char* WIFI_SSID     = "KJAH_201";       // ← Change this
const char* WIFI_PASSWORD = "36140343";   // ← Change this

// ── KuetTrack Server ─────────────────────────────────────────
const char* SERVER_IP    = "192.168.1.100";  // ← Your server LAN IP
const int  SERVER_PORT  = 5000;
const char* STATION_ID  = "STATION_01";     // ← Unique per station
const char* BIKE_ID     = "VT-023";         // ← Bike attached to this station

// ── Pin Definitions ──────────────────────────────────────────
#define RFID_SS_PIN    21   // SDA/CS
#define RFID_RST_PIN   22   // RST
#define LED_GREEN      2    // Green LED
#define LED_RED        4    // Red LED
#define BUZZER_PIN     5    // Active buzzer
#define SERVO_PIN      15   // Servo motor
#define LCD_SDA        26   // I2C SDA
#define LCD_SCL        27   // I2C SCL

// ── Hardware Objects ─────────────────────────────────────────
MFRC522 mfrc522(RFID_SS_PIN, RFID_RST_PIN);
LiquidCrystal_I2C lcd(0x27, 16, 2);  // Change 0x27 to 0x3F if needed
Servo lockServo;

// ── State Variables ──────────────────────────────────────────
bool bikeUnlocked = false;
String activeRideId = "";
String activeUserRfid = "";
unsigned long lastScanTime = 0;
const unsigned long SCAN_COOLDOWN = 2000; // 2s between scans

// ================================================================
//  SETUP
// ================================================================
void setup() {
  Serial.begin(115200);
  Serial.println("\n🚲 KuetTrack Station Booting...");

  // Init pins
  pinMode(LED_GREEN, OUTPUT);
  pinMode(LED_RED,   OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(LED_RED, HIGH);  // Red on = locked

  // Init LCD
  Wire.begin(LCD_SDA, LCD_SCL);
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0,0);  lcd.print("KuetTrack v1.0");
  lcd.setCursor(0,1);  lcd.print("Connecting WiFi");

  // Init Servo (locked position = 0°)
  lockServo.attach(SERVO_PIN);
  lockServo.write(0);  // 0° = locked

  // Connect WiFi
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("Connecting to WiFi");
  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 20) {
    delay(500);
    Serial.print(".");
    tries++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n✅ WiFi Connected! IP: " + WiFi.localIP().toString());
    lcd.clear();
    lcd.setCursor(0,0);  lcd.print("WiFi Connected!");
    lcd.setCursor(0,1);  lcd.print(WiFi.localIP());
    delay(2000);
  } else {
    Serial.println("\n❌ WiFi Failed! Running offline.");
    lcd.clear();
    lcd.print("WiFi FAILED!");
  }

  // Init RFID
  SPI.begin();
  mfrc522.PCD_Init();
  mfrc522.PCD_DumpVersionToSerial();
  Serial.println("💳 RFID reader ready. Waiting for cards...");

  lcd.clear();
  lcd.setCursor(0,0);  lcd.print("  KuetTrack 🚲  ");
  lcd.setCursor(0,1);  lcd.print("Tap card to ride");
}

// ================================================================
//  MAIN LOOP
// ================================================================
void loop() {
  // Debounce: ignore scans within cooldown window
  if (millis() - lastScanTime < SCAN_COOLDOWN) return;

  // Wait for RFID card
  if (!mfrc522.PICC_IsNewCardPresent() || !mfrc522.PICC_ReadCardSerial()) return;
  lastScanTime = millis();

  // Read UID bytes → hex string "A3:4F:2B:11"
  String uid = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    if (i > 0) uid += ":";
    if (mfrc522.uid.uidByte[i] < 0x10) uid += "0";
    uid += String(mfrc522.uid.uidByte[i], HEX);
  }
  uid.toUpperCase();
  Serial.println("💳 Card tapped: " + uid);

  handleCardTap(uid);

  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();
}

// ================================================================
//  CARD TAP HANDLER — Routes to unlock or lock
// ================================================================
void handleCardTap(String uid) {
  lcd.clear();
  lcd.print("Verifying card...");

  if (!bikeUnlocked) {
    // Bike is locked — try to authenticate
    authenticateRfid(uid);
  } else {
    if (uid == activeUserRfid) {
      // Same user tapping again — end ride
      endRide(uid);
    } else {
      // Different card while bike is in use
      Serial.println("⚠️ Bike in use by another user");
      lcd.clear();
      lcd.print("Bike in use!");
      shortBeep(3);
    }
  }
}

// ================================================================
//  AUTHENTICATE — Calls GET /api/auth/user-by-rfid/:uid
// ================================================================
void authenticateRfid(String uid) {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("❌ No WiFi — cannot authenticate");
    denyAccess("No WiFi!");
    return;
  }

  String url = String("http://") + SERVER_IP + ":" +
               String(SERVER_PORT) + "/api/auth/user-by-rfid/" + uid;

  Serial.println("🌐 Calling: " + url);

  HTTPClient http;
  http.begin(url);
  http.setTimeout(8000);
  int httpCode = http.GET();

  Serial.println("📡 HTTP Status: " + String(httpCode));

  if (httpCode == 200) {
    String payload = http.getString();
    Serial.println("✅ Response: " + payload);

    StaticJsonDocument<512> doc;
    if (deserializeJson(doc, payload) == DeserializationError::Ok) {
      bool authorized = doc["authorized"];
      if (authorized) {
        const char* firstName    = doc["user"]["firstName"];
        const char* plan         = doc["user"]["plan"];
        float       walletBal    = doc["user"]["walletBalance"];

        grantAccess(firstName, plan, walletBal);
        startRide(uid);
      } else {
        denyAccess("Not Authorized");
      }
    }
  } else if (httpCode == 404) {
    Serial.println("❌ Card not registered");
    denyAccess("Unknown Card!");
  } else {
    Serial.println("❌ Server error: " + String(httpCode));
    denyAccess("Server Error!");
  }
  http.end();
}

// ================================================================
//  START RIDE — Calls POST /api/rides/start
// ================================================================
void startRide(String uid) {
  HTTPClient http;
  String url = String("http://") + SERVER_IP + ":" + String(SERVER_PORT) + "/api/rides/start";
  http.begin(url);
  http.addHeader("Content-Type", "application/json");

  StaticJsonDocument<128> doc;
  doc["rfidUid"]   = uid;
  doc["stationId"] = STATION_ID;
  doc["bikeId"]    = BIKE_ID;
  String body;
  serializeJson(doc, body);

  int code = http.POST(body);
  if (code == 200 || code == 201) {
    StaticJsonDocument<256> resp;
    deserializeJson(resp, http.getString());
    activeRideId = resp["rideId"].as<String>();
    Serial.println("🚲 Ride started. ID: " + activeRideId);
  }
  http.end();
}

// ================================================================
//  END RIDE — Calls POST /api/rides/end
// ================================================================
void endRide(String uid) {
  Serial.println("🏁 Ending ride for: " + uid);
  lcd.clear();
  lcd.print("Ending ride...");

  HTTPClient http;
  String url = String("http://") + SERVER_IP + ":" + String(SERVER_PORT) + "/api/rides/end";
  http.begin(url);
  http.addHeader("Content-Type", "application/json");

  StaticJsonDocument<128> doc;
  doc["rfidUid"]   = uid;
  doc["rideId"]    = activeRideId;
  doc["stationId"] = STATION_ID;
  String body;
  serializeJson(doc, body);

  int code = http.POST(body);
  if (code == 200) {
    StaticJsonDocument<256> resp;
    deserializeJson(resp, http.getString());
    float fare = resp["fare"];
    String duration = resp["duration"].as<String>();

    lcd.clear();
    lcd.setCursor(0,0); lcd.print("Fare: " + String(fare) + " BDT");
    lcd.setCursor(0,1); lcd.print(duration);
    Serial.println("✅ Ride ended. Fare: " + String(fare));
  }
  http.end();

  lockBike();
  activeRideId = "";
  activeUserRfid = "";
  bikeUnlocked = false;

  delay(3000);
  lcd.clear();
  lcd.setCursor(0,0); lcd.print("  KuetTrack 🚲  ");
  lcd.setCursor(0,1); lcd.print("Tap card to ride");
}

// ================================================================
//  ACCESS GRANT — Green LED, unlock servo, display user
// ================================================================
void grantAccess(const char* name, const char* plan, float balance) {
  Serial.println("✅ ACCESS GRANTED → " + String(name));

  digitalWrite(LED_RED,   LOW);
  digitalWrite(LED_GREEN, HIGH);
  shortBeep(1);
  unlockBike();

  lcd.clear();
  lcd.setCursor(0,0); lcd.print("Hi, " + String(name));
  lcd.setCursor(0,1); lcd.print(String(plan) + " | " + String((int)balance) + "BDT");
}

// ================================================================
//  ACCESS DENY — Red LED blink, buzzer long
// ================================================================
void denyAccess(String reason) {
  Serial.println("❌ ACCESS DENIED: " + reason);

  lcd.clear();
  lcd.print("Access Denied!");
  lcd.setCursor(0,1); lcd.print(reason);

  for (int i = 0; i < 3; i++) {
    digitalWrite(LED_RED, HIGH);
    digitalWrite(BUZZER_PIN, HIGH);
    delay(200);
    digitalWrite(LED_RED, LOW);
    digitalWrite(BUZZER_PIN, LOW);
    delay(200);
  }
  digitalWrite(LED_RED, HIGH); // Keep red on = locked

  delay(2000);
  lcd.clear();
  lcd.setCursor(0,0); lcd.print("  KuetTrack 🚲  ");
  lcd.setCursor(0,1); lcd.print("Tap card to ride");
}

// ================================================================
//  SERVO LOCK / UNLOCK
// ================================================================
void unlockBike() {
  lockServo.write(90);  // 90° = unlocked
  bikeUnlocked = true;
  Serial.println("🔓 Bike unlocked");
}
void lockBike() {
  lockServo.write(0);   // 0° = locked
  digitalWrite(LED_GREEN, LOW);
  digitalWrite(LED_RED, HIGH);
  Serial.println("🔒 Bike locked");
}

// ================================================================
//  HELPER: Short beep(s)
// ================================================================
void shortBeep(int count) {
  for (int i = 0; i < count; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    delay(100);
    digitalWrite(BUZZER_PIN, LOW);
    if (count > 1) delay(100);
  }
}
📝 Before Uploading — Checklist
✅ Change WIFI_SSID and WIFI_PASSWORD to your network
✅ Set SERVER_IP to your backend server's LAN IP address
✅ Set a unique STATION_ID and BIKE_ID for each physical station
✅ In Arduino IDE: Board = ESP32 Dev Module, Upload Speed = 921600
✅ If LCD doesn't show text, try changing 0x27 to 0x3F in the I2C address
✅ Open Serial Monitor at 115200 baud to see debug logs
